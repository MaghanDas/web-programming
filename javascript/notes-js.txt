document.addEventListener("keydown", (e) => {
    let keys = "";
    if (e.ctrlKey) keys += "Ctrl + ";
    if (e.shiftKey) keys += "Shift + ";
    if (e.altKey) keys += "Alt + ";
    keys += e.key;
});
